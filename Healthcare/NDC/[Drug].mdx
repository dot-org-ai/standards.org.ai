  code={drug.productNDC}

/>

## NDC Code

**{drug.productNDC}**

## Drug Names

<div className="drug-names">
  <div className="proprietary-name">
    <strong>Proprietary (Brand) Name:</strong> {drug.proprietaryName}
  </div>
  {drug.nonProprietaryName && (
    <div className="non-proprietary-name">
      <strong>Non-Proprietary (Generic) Name:</strong> {drug.nonProprietaryName}
    </div>
  )}
</div>

## Active Ingredients

{drug.activeIngredient && drug.activeIngredient.length > 0 && (
  <div className="active-ingredients">
    <ul>
      {drug.activeIngredient.map((ingredient, idx) => (
        <li key={idx}>
          <strong>{ingredient}</strong>
          {drug.strengthNumber && drug.strengthNumber[idx] && (
            <span> - {drug.strengthNumber[idx]} {drug.strengthUnit && drug.strengthUnit[idx]}</span>
          )}
        </li>
      ))}
    </ul>
  </div>
)}

## Dosage Information

<div className="dosage-info">
  {drug.dosageForm && (
    <div className="dosage-form">
      <strong>Dosage Form:</strong> {drug.dosageForm}
    </div>
  )}
  {drug.routeOfAdministration && drug.routeOfAdministration.length > 0 && (
    <div className="route">
      <strong>Route of Administration:</strong> {drug.routeOfAdministration.join(', ')}
    </div>
  )}
</div>

## Labeler

{drug.labeler && (
  <LabelerLink code={drug.labeler} />
)}
