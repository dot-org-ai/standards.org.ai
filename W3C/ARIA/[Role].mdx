    function trapFocus(containerElement) {
      const focusableElements = containerElement.querySelectorAll(

      const firstElement = focusableElements[0]

## Common Mistakes

### Mistake 1: Missing Required Attributes

```html
<!-- Incorrect: Missing required attributes -->
<div role="{role.name}">
  Content
</div>

<!-- Correct: All required attributes present -->
<div
  role="{role.name}"
  {role.accessibleNameRequired && 'aria-label="Descriptive name"'}
  {role.requiredStates.map(s => `${s.name}="${s.defaultValue}"`).join('\n  ')}
>
  Content
</div>
```

### Mistake 2: Incorrect Context

{role.requiredContext.length > 0 && `
\`\`\`html
<!-- Incorrect: Wrong parent context -->
<div role="navigation">
  <div role="${role.name}">
    Content
  </div>
</div>

<!-- Correct: Proper parent context -->
<div role="${role.requiredContext[0]}">
  <div role="${role.name}">
    Content
  </div>
</div>
\`\`\`
`}

### Mistake 3: Using When Native HTML Exists

```html
<!-- Incorrect: Using role when native HTML available -->
<div role="button" tabindex="0" onclick="doSomething()">
  Click me
</div>

<!-- Correct: Use native HTML element -->
<button type="button" onclick="doSomething()">
  Click me
</button>
```

### Mistake 4: Overriding Implicit Semantics

```html
<!-- Incorrect: Overriding native semantics unnecessarily -->
<button role="link">
  This is confusing
</button>

<!-- Correct: Use appropriate element -->
<a href="/page">
  This is a link
</a>
```
