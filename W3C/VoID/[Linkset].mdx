<section className="sparql-queries">
  <h3>SPARQL Query Examples</h3>

  <h4>Get linkset details</h4>
  <pre><code>{`PREFIX void: <http://rdfs.org/ns/void#>
PREFIX dcterms: <http://purl.org/dc/terms/>

SELECT ?title ?subjectsTarget ?objectsTarget ?triples
WHERE {
  <${linkset.uri}> dcterms:title ?title ;
                   void:subjectsTarget ?subjectsTarget ;
                   void:objectsTarget ?objectsTarget ;
                   void:triples ?triples .
}`}</code></pre>

  <h4>Find all link predicates used</h4>
  <pre><code>{`PREFIX void: <http://rdfs.org/ns/void#>

SELECT ?predicate
WHERE {
  <${linkset.uri}> void:linkPredicate ?predicate .
}`}</code></pre>

  <h4>Query actual links in the linkset</h4>
  <pre><code>{`# Assuming you have access to the actual data
PREFIX void: <http://rdfs.org/ns/void#>

SELECT ?subject ?predicate ?object
WHERE {
  <${linkset.uri}> void:subjectsTarget ?subjectsDataset ;
                   void:objectsTarget ?objectsDataset ;
                   void:linkPredicate ?predicate .

  # Query the actual triples
  ?subject ?predicate ?object .

  # Filter to only include links matching the linkset definition
  FILTER(STRSTARTS(STR(?subject), ?subjectsUriSpace))
  FILTER(STRSTARTS(STR(?object), ?objectsUriSpace))
}
LIMIT 100`}</code></pre>
</section>
