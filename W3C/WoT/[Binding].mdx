## Implementation Examples

### HTTP Binding Example

```javascript
// Read property using HTTP GET
const response = await fetch('{binding.href}', {
  method: '{binding.htv_methodName}',
  headers: {
    'Accept': '{binding.contentType}',
    'Authorization': 'Bearer <token>'
  }
});
const data = await response.json();
console.log('Data:', data);
```

```javascript
// Write property using HTTP PUT
const response = await fetch('{binding.href}', {
  method: 'PUT',
  headers: {
    'Content-Type': '{binding.contentType}',
    'Authorization': 'Bearer <token>'
  },
  body: JSON.stringify(newValue)
});
```

### CoAP Binding Example

```javascript
// Using node-coap library
const coap = require('coap');

// CoAP GET request
const req = coap.request({
  hostname: 'coap.example.com',
  pathname: '/thing/property',
  method: '{binding.cov_method}',
  confirmable: true
});

req.on('response', (res) => {
  console.log('Response:', res.payload.toString());
});

req.end();
```

```javascript
// CoAP Observe
const req = coap.request({
  hostname: 'coap.example.com',
  pathname: '/thing/property',
  observe: true
});

req.on('response', (res) => {
  console.log('Update:', res.payload.toString());
});

req.end();
```

### MQTT Binding Example

```javascript
// Using MQTT.js library
const mqtt = require('mqtt');
const client = mqtt.connect('mqtt://broker.example.com');

client.on('connect', () => {
  // Subscribe to property updates
  client.subscribe('{binding.mqv_topic}', { qos: {binding.mqv_qos} });

  // Publish new value
  client.publish('{binding.mqv_topic}/set',
    JSON.stringify(newValue),
    {
      qos: {binding.mqv_qos},
      retain: {binding.mqv_retain}
    }
  );
});

client.on('message', (topic, message) => {
  console.log('Received:', message.toString());
});
```

### WebSocket Binding Example

```javascript
// WebSocket connection
const ws = new WebSocket('wss://thing.example.com/ws');

ws.onopen = () => {
  console.log('Connected');

  // Send request
  ws.send(JSON.stringify({
    action: 'readProperty',
    name: 'temperature'
  }));
};

ws.onmessage = (event) => {
  const data = JSON.parse(event.data);
  console.log('Response:', data);
};
```

### Modbus Binding Example

```javascript
// Using node-modbus library
const ModbusRTU = require('modbus-serial');
const client = new ModbusRTU();

// Connect to Modbus TCP
await client.connectTCP('{binding.href}');
client.setID(1);

// Read holding registers
const data = await client.readHoldingRegisters(
  {binding.modv_address},
  1
);
console.log('Value:', data.data[0]);
```
