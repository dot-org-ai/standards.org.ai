    <h4>Key Properties</h4>
    <ul>
      <li><strong>foaf:maker:</strong> The person or agent who created the document</li>

    </ul>
    <h4>Use Cases</h4>
    <ul>

<${document.uri}> a foaf:Document ;
  dc:title "${document.title}" ;${document.description ? `

  dc:created "${document.created}" ;` : ''}${document.modified ? `
  dc:modified "${document.modified}" ;` : ''}
  .`}</code></pre>
</section>

<section className="sparql-queries">
  <h3>SPARQL Query Examples</h3>

  <h4>Find all documents by maker</h4>
  <pre><code>{`PREFIX foaf: <http://xmlns.com/foaf/0.1/>
PREFIX dc: <http://purl.org/dc/elements/1.1/>

SELECT ?document ?title
WHERE {
  ?document a foaf:Document ;
            foaf:maker <${document.maker?.uri || 'http://example.org/person'}> ;
            dc:title ?title .
}
ORDER BY DESC(?created)`}</code></pre>

  <h4>Find documents about a topic</h4>
  <pre><code>{`PREFIX foaf: <http://xmlns.com/foaf/0.1/>
PREFIX dc: <http://purl.org/dc/elements/1.1/>

SELECT ?document ?title ?maker
WHERE {
  ?document a foaf:Document ;
            foaf:topic <${document.topic?.[0]?.uri || 'http://example.org/topic'}> ;
            dc:title ?title .
  OPTIONAL { ?document foaf:maker ?maker }
}`}</code></pre>

  <h4>Find primary topic document</h4>
  <pre><code>{`PREFIX foaf: <http://xmlns.com/foaf/0.1/>

SELECT ?topic ?description
WHERE {
  <${document.uri}> foaf:primaryTopic ?topic .
  OPTIONAL { ?topic foaf:isPrimaryTopicOf <${document.uri}> }
}`}</code></pre>
</section>

  <h4>Create a FOAF document</h4>
  <pre><code>{`@prefix foaf: <http://xmlns.com/foaf/0.1/> .

             <http://dbpedia.org/resource/SPARQL> .`}</code></pre>

  <h4>Link person to their documents</h4>

  foaf:made <http://example.org/document/my-article> ,
            <http://example.org/document/another-article> .`}</code></pre>
</section>
