## Usage Examples

<CodeExample
  title={`Reading ${property.name}`}
  language="javascript"
  code={property.examples?.read}
/>

{!property.readOnly && property.examples?.write && (
  <CodeExample
    title={`Setting ${property.name}`}
    language="javascript"
    code={property.examples.write}
  />
)}

{property.examples?.advanced && (
  <CodeExample
    title="Advanced usage"
    language="javascript"
    code={property.examples.advanced}
  />
)}

{property.examples?.observer && (
  <CodeExample
    title="Observing changes"
    language="javascript"
    code={property.examples.observer}
  />
)}

## Common Use Cases

{property.useCases?.map((useCase, index) => (
  <div key={index} className="use-case">
    <h3>{useCase.title}</h3>
    <p>{useCase.description}</p>
    {useCase.example && (
      <CodeExample
        language="javascript"
        code={useCase.example}
      />
    )}
  </div>
))}

## Type Information

{property.typeDetails && (
  <div className="type-details">
    <p><strong>Type:</strong> {property.type}</p>
    {property.typeDetails.description && (
      <p>{property.typeDetails.description}</p>
    )}
    {property.typeDetails.possibleValues && (
      <>
        <h3>Possible Values</h3>
        <ul>
          {property.typeDetails.possibleValues.map((value, index) => (
            <li key={index}>
              <code>{value.value}</code>: {value.description}
            </li>
          ))}
        </ul>
      </>
    )}
  </div>
)}

## Notes

{property.notes?.map((note, index) => (
  <div key={index} className="note">
    <p>{note}</p>
  </div>
))}

{property.readOnly && (
  <div className="readonly-notice">
    This property is read-only and cannot be modified directly.
  </div>
)}
