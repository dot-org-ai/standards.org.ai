import { Header } from '@/components/standards/Header'
import { Description } from '@/components/standards/Description'
import { MethodSignature } from '@/components/w3c/dom/MethodSignature'
import { ParametersTable } from '@/components/w3c/dom/ParametersTable'
import { ReturnValue } from '@/components/w3c/dom/ReturnValue'
import { ExceptionsTable } from '@/components/w3c/dom/ExceptionsTable'
import { RelatedMethods } from '@/components/w3c/dom/RelatedMethods'
import { CodeExample } from '@/components/standards/CodeExample'
import { BrowserCompatibility } from '@/components/standards/BrowserCompatibility'
import { SpecificationLinks } from '@/components/standards/SpecificationLinks'

<Header
  type="DOM Method"
  code={method.name}
  name={`${method.interface}.${method.name}()`}
  standard="W3C DOM"
/>

<Description content={method.description} />

## Method Signature

<MethodSignature
  method={method.name}
  interface={method.interface}
  parameters={method.parameters}
  returnType={method.returnType}
  static={method.static}
  async={method.async}
/>

## Parameters

<ParametersTable
  parameters={method.parameters}
/>

## Return Value

<ReturnValue
  type={method.returnType}
  description={method.returnDescription}
/>

{method.throws && method.throws.length > 0 && (
  <>
    ## Exceptions

    <ExceptionsTable
      exceptions={method.throws}
    />
  </>
)}

## Usage Examples

<CodeExample
  title={`Basic usage of ${method.name}()`}
  language="javascript"
  code={method.examples?.basic}
/>

{method.examples?.withParameters && method.parameters.length > 0 && (
  <CodeExample
    title="With parameters"
    language="javascript"
    code={method.examples.withParameters}
  />
)}

{method.examples?.errorHandling && method.throws && (
  <CodeExample
    title="Error handling"
    language="javascript"
    code={method.examples.errorHandling}
  />
)}

{method.examples?.advanced && (
  <CodeExample
    title="Advanced usage"
    language="javascript"
    code={method.examples.advanced}
  />
)}

## Common Use Cases

{method.useCases?.map((useCase, index) => (
  <div key={index} className="use-case">
    <h3>{useCase.title}</h3>
    <p>{useCase.description}</p>
    {useCase.example && (
      <CodeExample
        language="javascript"
        code={useCase.example}
      />
    )}
  </div>
))}

## Notes

{method.notes?.map((note, index) => (
  <div key={index} className="note">
    <p>{note}</p>
  </div>
))}

## Related Methods

<RelatedMethods
  interface={method.interface}
  currentMethod={method.name}
  related={method.relatedMethods}
/>

<BrowserCompatibility
  feature={`api.${method.interface}.${method.name}`}
  standard="W3C DOM"
/>

<SpecificationLinks
  specifications={method.specifications}
  standard="DOM"
/>
