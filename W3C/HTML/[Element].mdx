import { Header } from '@/components/standards/Header'
import { Description } from '@/components/standards/Description'

  type="HTML Element"
  code={element.tag}
  name={element.name}
  standard="W3C HTML"
/>

)}

<AttributesList

  title={`Basic ${element.name} usage`}
  language="html"

  <CodeExample
    title="Advanced usage"
    language="html"
    code={element.examples.advanced}
  />
)}

{element.examples?.withJavaScript && (
  <CodeExample
    title="Using with JavaScript"
    language="javascript"
    code={element.examples.withJavaScript}
  />
)}
