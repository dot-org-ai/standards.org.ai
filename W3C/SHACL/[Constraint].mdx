{constraint.type === 'sh:minCount' && (
  <section className="details">
    <h3>Detailed Information</h3>
    <p>
      <strong>sh:minCount</strong> specifies the minimum number of value nodes that satisfy
      the property shape. The constraint is satisfied if the number of value nodes is
      greater than or equal to the specified minimum.
    </p>
    <h4>Common Values</h4>
    <ul>
      <li><code>sh:minCount 0</code> - Optional property (default)</li>
      <li><code>sh:minCount 1</code> - Required property (at least one value)</li>
      <li><code>sh:minCount 2</code> - At least two values required</li>
    </ul>
  </section>
)}

{constraint.type === 'sh:maxCount' && (
  <section className="details">
    <h3>Detailed Information</h3>
    <p>
      <strong>sh:maxCount</strong> specifies the maximum number of value nodes that satisfy
      the property shape. The constraint is satisfied if the number of value nodes is
      less than or equal to the specified maximum.
    </p>
    <h4>Common Values</h4>
    <ul>
      <li><code>sh:maxCount 1</code> - Single-valued property (at most one value)</li>
      <li><code>sh:maxCount 5</code> - Up to five values allowed</li>
    </ul>
  </section>
)}

{constraint.type === 'sh:datatype' && (
  <section className="details">
    <h3>Detailed Information</h3>
    <p>
      <strong>sh:datatype</strong> constrains the datatype of literal value nodes. Values
      must have the specified datatype.
    </p>
    <h4>Common Datatypes</h4>
    <ul>
      <li><code>xsd:string</code> - Text strings</li>
      <li><code>xsd:integer</code> - Whole numbers</li>
      <li><code>xsd:decimal</code> - Decimal numbers</li>
      <li><code>xsd:boolean</code> - True/false values</li>
      <li><code>xsd:date</code> - Dates (YYYY-MM-DD)</li>
      <li><code>xsd:dateTime</code> - Date and time with timezone</li>
      <li><code>xsd:anyURI</code> - URI references</li>
    </ul>
  </section>
)}

{constraint.type === 'sh:class' && (
  <section className="details">
    <h3>Detailed Information</h3>
    <p>
      <strong>sh:class</strong> constrains value nodes to be instances of a specified class.
      Each value must have rdf:type of the specified class or a subclass.
    </p>
  </section>
)}

{constraint.type === 'sh:nodeKind' && (
  <section className="details">
    <h3>Detailed Information</h3>
    <p>
      <strong>sh:nodeKind</strong> constrains the type of RDF nodes that are allowed as values.
    </p>
    <h4>Node Kinds</h4>
    <ul>
      <li><code>sh:IRI</code> - Must be an IRI</li>
      <li><code>sh:BlankNode</code> - Must be a blank node</li>
      <li><code>sh:Literal</code> - Must be a literal</li>
      <li><code>sh:BlankNodeOrIRI</code> - Blank node or IRI</li>
      <li><code>sh:BlankNodeOrLiteral</code> - Blank node or literal</li>
      <li><code>sh:IRIOrLiteral</code> - IRI or literal</li>
    </ul>
  </section>
)}

{constraint.type === 'sh:pattern' && (
  <section className="details">
    <h3>Detailed Information</h3>
    <p>
      <strong>sh:pattern</strong> constrains string values to match a regular expression.
      Optionally use <code>sh:flags</code> for pattern matching flags.
    </p>
    <h4>Example Patterns</h4>
    <ul>
      <li><code>^[A-Z]</code> - Must start with uppercase letter</li>
      <li><code>\\d{"{3}"}-\\d{"{2}"}-\\d{"{4}"}</code> - SSN format (XXX-XX-XXXX)</li>
      <li><code>^https?://</code> - Must be HTTP or HTTPS URL</li>
      <li><code>@[a-z]{"{2}"}</code> - Must end with language tag</li>
    </ul>
  </section>
)}

{constraint.type === 'sh:minLength' && (
  <section className="details">
    <h3>Detailed Information</h3>
    <p>
      <strong>sh:minLength</strong> constrains the minimum length of string values.
      The length is counted in characters.
    </p>
  </section>
)}

{constraint.type === 'sh:maxLength' && (
  <section className="details">
    <h3>Detailed Information</h3>
    <p>
      <strong>sh:maxLength</strong> constrains the maximum length of string values.
      The length is counted in characters.
    </p>
  </section>
)}

{constraint.type === 'sh:in' && (
  <section className="details">
    <h3>Detailed Information</h3>
    <p>
      <strong>sh:in</strong> constrains values to be members of a specified list.
      Values must exactly match one of the items in the list.
    </p>
    <h4>Example</h4>
    <pre><code>{`sh:property [
  sh:path ex:status ;
  sh:in ("draft" "published" "archived") ;
] .`}</code></pre>
  </section>
)}
