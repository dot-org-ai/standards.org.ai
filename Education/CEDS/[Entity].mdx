                {attr.name}
              </a>
            </td>
            <td><code>{attr.dataType}</code></td>
            <td>{attr.required ? 'Yes' : 'No'}</td>
            <td>{attr.definition}</td>

      </tbody>

) : (
  <AttributesByEntity entity={entity.code} />
